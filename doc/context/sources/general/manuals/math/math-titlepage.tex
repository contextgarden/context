% language=us runpath=texruns:manuals/math

\startcomponent math-titlepage

\startMPpage

    StartPage ;
        fill Page withcolor if contextlmtxmode : .35blue else : .35magenta fi ;
        picture p ; p := image ( draw outlinetext.b
            ("\textdollar")
            (withcolor .35green)
            (withcolor .5yellow withpen pencircle scaled .1) ;
        ) ;
        if contextlmtxmode : p := p xscaled -1 fi ;
        p := p xysized(PaperWidth - 20pt,PaperHeight - 20pt ) ;
        p := p shifted - llcorner p shifted (10pt,10pt) ;
        draw p ;
        picture p ; p := image ( draw outlinetext.b
            ("math")
            (withcolor .35green)
            (withcolor .5yellow withpen pencircle scaled .2) ;
        ) ;
        p := p xsized(PaperWidth/2 - 60pt) ;
        p := p shifted - llcorner p shifted (10pt,10pt) ;
        draw p shifted (PaperWidth/2 + 40pt,10pt) ;
    StopPage ;

\stopMPpage

\page[empty] \setuppagenumber[number=1]

\stopcomponent
