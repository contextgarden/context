\startluacode
local context = context

local NC = context.NC
local NR = context.NR

interfaces.implement {
    name      = "ShowSyntax",
    public    = true,
    protected = true,
    arguments = "string",
    actions   = function(s)
        local t = chemistry.syntax[s]
     -- inspect(t)
        context.startcolumns { n = 4 }
            context.starttabulate { "|Tl|l|" }
                for k, v in table.sortedhash(t.keys) do
                    NC() context(k)
                    NC() context(v)
                    NC() NR()
                end
            context.stoptabulate()
        context.stopcolumns()
        context.blank { "2*big" }
    end
}
\stopluacode

\starttexdefinition tolerant protected ShowStructure[#1]#*[#2]#*[#3]#*[#4]
    \startlinecorrection
    \hbox \bgroup
        \setupchemicalframed
            [frame=on]
        \setupchemical
            [scale=big,
             size=small]
        \setbox\scratchboxone\ruledhbox to .25tw \bgroup
            \startchemical
                \chemical[#2][#3]
            \stopchemical
            \hss
        \egroup
        \setbox\scratchboxtwo\vbox to \htdp\scratchboxone\bgroup
            \hsize{1tw-\wd\scratchboxone-4\emwidth}
            \strut\tttf [#2] \ifparameter#3\or[#3]\fi
            \vfill
        \egroup
        \setbox\scratchboxthree\vbox to \htdp\scratchboxone\bgroup
            \hsize{3\emwidth}
            \strut\tttf \ignorespaces#1\removeunwantedspaces
            \vfill
        \egroup
        \dontleavehmode
        \box\scratchboxthree
        \hkern\emwidth
        \box\scratchboxone
        \hkern\emwidth
        \box\scratchboxtwo
    \egroup
    \stoplinecorrection
\stoptexdefinition
